//src\pages\activity\home\cloudDots.js
const allKeyArray = [

this.state = {
    allKeyArray:allKeyArray
}
render() {
    return (
        <View style={[styles.slippage]}>
            {this.state.allKeyArray.map((item, i) => {
                return (
                    // 内层小圆
                    <TouchableWithoutFeedback onPress={() => {
                        this.changeBG(i)
                        }} >
                    </TouchableWithoutFeedback>
                )
            })}
        </View>
    );
}

改变小球颜色并弹窗询问是否呼叫柜员。

//src\pages\activity\home\cloudDots.js
changeBG(k){
    let newAllKeyArray = this.state.allKeyArray;
    for(var i = 0;i<newAllKeyArray.length;i++){
        if(k == i){
            newAllKeyArray[i].itemBg = itemBgYellow;
        }else{
            newAllKeyArray[i].itemBg = itemBg;
        }
    }
    this.callCountModel(k)
}

用户确定后开始请求视频流程。

//src\pages\activity\home\cloudDots.js
callCountModel(k){
    let defaultAction = [
        {
          text: '取消',
          onPress: () => {
            var newAllKeyArray = this.state.allKeyArray;
            for(var i = 0;i<newAllKeyArray.length;i++){
                newAllKeyArray[i].itemBg = itemBg;
            }
          },
          style: 'default',
        },
        {
          text: '确定', onPress: () => {
            this.callCount();
          },
        }
    ];
  window.$Modal.confirm('呼叫柜员','确认呼叫远程柜员吗?',defaultAction)
}

# AnyChat.init 初始化

在确认发起视频之后，需要先（AnyChat.init）初始化AnyChat。随后在成功回调中(AnyChat.enterLine)申请用户进线。

callCount() {
        AnyChat.init(( succ )=>{
            console.warn( '初始化成功回调',succ )
            let businessId = ''
            let params = {
                enterLineUrl: enterLineUrl,
                businessId: businessId,
                IDNO:this.state.userInfoData.idCard,
                IDTYPE:'02',
                IDNAME:window.userName
            }
            AnyChat.enterLine ( params, ( succ )=>{
                console.warn( '进线成功回调='+ succ )
                router.load("tellerOperation",{
                    account:"liudabin",
                });
            }, ( err )=>{
                console.warn('进线失败', err )
                $Toast.info('呼叫视频服务失败')
                this.unCallCount();
            });
        }, ( err )=>{
            console.warn( err )
            $Toast.info('呼叫视频服务失败')
            this.unCallCount();
        });
}
unCallCount(){
    AnyChat.quitLine((succ)=>{
        console.warn('succ',succ)
    },( err )=>{
        console.warn('err',err)
    })
    AnyChat.loginOut(( succ )=>{
        console.warn( '销毁后=' + succ );
    },( err )=>{
        console.warn();
    });
}
