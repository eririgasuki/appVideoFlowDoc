# App.js 项目入口

手机银行app的入口文件为工程根目录下的app.js

路由的入口组件为<AppTag>,由react-navigation创建。

import { View, StatusBar } from 'react-native';
import { createAppContainer } from 'react-navigation';
import RootStack from './router/router.js';
let AppTag = createAppContainer(RootStack);
render() {
  return (
    <View style={{ width: '100%', height: '100%' }}>
      <StatusBar
        animated={true} //指定状态栏的变化是否应以动画形式呈现。目前支持这几种样式：backgroundColor,barStyle和hidden
        hidden={false} //是否隐藏状态栏。
        backgroundColor="transparent" //状态栏的背景色
        translucent={true} //指定状态栏是否透明。设置为true时，应用会在状态栏之下绘制（即所谓“沉浸式”——状态栏遮住一部分）。常和带有半透明背景色的状态栏搭配使用。
        barStyle="light-content"
      />
      <AppTag />
    </View>
  );
}

路由配置文件为router.js。

//router\router.js
import {createStackNavigator,createAppContainer} from 'react-navigation';
var homeScrollable = require("$/pages/activity/home/homeScrollable.js")
let RootStack = createStackNavigator(
    {"homeScrollable":{"screen":homeScrollable,"componentName":"homeScrollable","navigationOptions":{"header":null}},{"initialRouteName": "homeScrollable"}
)

设置的初始路由（initialRouteName）页面为homeScrollable，即打开应用后显示的首页。

# homeScrollable.js 应用首页

在应用首页左下角导航栏（TabBottom组件）选择云网点进入请求视频页面。

import ScrollableTabView from 'react-native-scrollable-tab-view';
import { View } from 'react-native';
import TabBottom from './TabBottom';
const tabTitles = ['云网点', '首页', '金融', '我的'];
render() {
    return (
        <View style={{ width: screenWidth, height: '100%' }}>
            <ScrollableTabView
                initialPage={1}
                renderTabBar={() =>
                    <TabBottom
                        ref={ref => this.tab = ref}
                        {...this.props}
                        tabNames={tabTitles}
                    />
                }
            >
            </ScrollableTabView>
        </View>
    );
}

在TabBottom中可以看到,当点击第一个云网点时，会跳转cloudDots页面。

const router = require('$/router-control');
goJump = (i) => {
    if (i == '0') {
        router.load('cloudDots')
    }
}
render() {
    return (
        <View style={styles.tabs}>
            {this.props.tabs.map((tab, i) => {
                return (
                    <TouchableOpacity
                        key={i}
                        onPress={() => {this.goJump(i)}}>
                    </TouchableOpacity>
                )
            })}
        </View>
    );
}

# login.js 登录页面

第一次进入视频页面时,因为没有Token会需要登录。先跳转到login.js页面。

//src\router-control.js
function load(router, params, mainBundleFlag, bundle) {
  } else {
    if (!getLoginToken().data) {
      jumpToLogin(router, params, mainBundleFlag, bundle);
    } else {
      getCurrentRouter().navigate(router, params);
    }
  }
}

在login中获取Token后，跳转到目的地路由(destRouter)即cloudDots页面

//src\pages\activity\login\login.js
this.state = {
  username:  '17600169392',
  password: '123456',
  destRouter: props.navigation.state.params.params,
}
loginByPwd = () => {
  const data = {
    username: this.state.username,
    password: this.state.password
  };
  NetworkUtil.networkService('/oauth/token', data, (response) => {
    NetworkUtil.networkService('/user/info', {}, (resp) => {
      $Storage.save({
        key: 'userInfo',
        data: JSON.stringify(resp)
      });
      $Toast.success('登录成功！')
      window.userName = resp.cnName
      window.userPhone = resp.phone
      if (this.state.destRouter) {
        StatusBar.setBarStyle('light-content')
        router.replace(this.state.destRouter);
      } else {
        router.back();
      }
}

登录后的cloudDots页面即视频入口页面，接下来会介绍，从发起视频到完成交易挂断视频的流程。
